import "../main.scss";

import {
  drawText,
  drawCircle,
  createSvg,
  drawCircleProgressBar,
} from "../helpers/svg-functions";
import { Component } from "react";
import * as d3 from "d3";

/**
 * Pie chart of the progress.
 * @component
 * @extends Component
 */
class PieChart extends Component {
  /**
   * Create a circle progress bar.
   * @param   {number}  value   The decimal value of progress.
   * @param   {number}  width   The width of the chart.
   * @param   {number}  height  The height of the chart.
   */
  constructor(props) {
    super(props);
    this.value = props.value;
    this.width = 258;
    this.height = 264;
  }
  componentDidMount() {
    /** Delete the chart if it already exists. */
    d3.select("#piechart svg").remove();

    /**
     * Create the SVG container.
     */
    const svg = createSvg("#piechart", this.width, this.height);

    /**
     * Create the circle width the circle progress bar.
     */
    const piechart = svg.append("g").attr("class", "piechart");
    drawCircle(piechart, 129, 132, 85, "white");
    drawCircleProgressBar(piechart, 129, 132, 90, 10, this.value, "#f00");

    /**
     * Creates the title of the chart.
     */
    const title = svg.append("g").attr("class", "piechart-title");
    drawText(svg, 30, 30, "Score", "piechart-title", "#20253a");

    /**
     * Create the text inside the chart.
     */
    const texts = svg.append("g").attr("class", "piechart-text");
    drawText(
      texts,
      108,
      110,
      `${this.value * 100} %`,
      "piechart-value",
      "#282d30"
    );
    drawText(texts, 102, 143, "de votre", "piechart-text", "#74798c");
    drawText(texts, 103, 167, "objectif", "piechart-text", "#74798c");
  }
  render() {
    return <></>;
  }
}

export default PieChart;
